pm.test("Status 200", () => pm.response.to.have.status(200));
pm.test("Response is JSON", () => pm.response.to.be.json);

smallResponseTime = pm.response.responseTime;
pm.environment.set("smallResponseTime", smallResponseTime);
console.log("Small request time:", smallResponseTime, "ms");

// If previous largeResponseTime exists, compare
largeResponseTime = pm.environment.get("largeResponseTime");
if (largeResponseTime) {
    pm.test("Small response faster than large", () => {
        pm.expect(parseInt(smallResponseTime)).to.be.below(parseInt(largeResponseTime));
    });
}
